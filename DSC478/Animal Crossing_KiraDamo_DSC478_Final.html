def cosine_recommender(df, item, simMat, k=5):
    indices = pd.Series(range(len(df)), index=df.index)
    if item not in indices.index:
        matches = indices.index[indices.index.str.startswith(item)]
        if len(matches) > 0:
            item_name = matches[0] # Take the first variation (like '? Block (None)')
        else:
            return f"Item '{item}' not found in dataset.", []
    else:
        item_name = item
        
    item_idx = indices[item]
    sims = simMat[item_idx].flatten()
    idx = np.argsort(sims)[::-1]
    idx = idx[idx != item_idx]
    neigh_idx = idx[:k]
    neigh_sims = sims[neigh_idx]
    return df.index[neigh_idx].tolist(), neigh_sims

def euclidean_recommender(df, item, k=5):
    X = df.values
    knn = NearestNeighbors(n_neighbors=6, metric='euclidean', algorithm='auto')
    knn.fit(X)
    indices = pd.Series(range(len(df)), index=df.index)
    idx = indices[item]
    
    #reshape the row for the model
    query_row = X[idx].reshape(1, -1)
    
    #find the nearest distances and their indices
    distances, neighbor_indices = knn.kneighbors(query_row)
    
    #flatten the results and skip the first one (which is the item itself)
    neigh_idx = neighbor_indices.flatten()[1:]
    distances = distances.flatten()[1:]
    
    #get the names
    recs = df_model_ready.index[neigh_idx]
    
    return recs, distances

def ensemble(outputs, weights=None):
    if weights is None:
        weights = {name: 1/len(model_outputs) for name in model_outputs.keys()}

    scaler = MinMaxScaler()
    all_scores = []

    for i in range(len(outputs)):
        df_results = outputs[i].copy()

        scores = df_results[['scores']].values
        
        #scale to 0-1 range so models are comparable
        df_results['norm_score'] = scaler.fit_transform(scores)
        df_results['weighted_score'] = df_results['norm_score'] * weights[i]
        
        #keep only the item names and the new weighted score
        all_scores.append(df_results[['item', 'weighted_score']])

    #combine all model results and sum the weighted scores
    ensemble_df = pd.concat(all_scores).groupby('item')['weighted_score'].sum().reset_index()
    
    #sort by highest score for the final recommendation
    return ensemble_df.sort_values(by='weighted_score', ascending=False)
